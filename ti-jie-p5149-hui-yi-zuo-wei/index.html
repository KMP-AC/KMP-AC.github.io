在此，先简单介绍一下归并排序：

归并排序由三个部分组成：分解、合并，归返（其实是我起的qwq）

归并排序是一种分治算法，时间复杂度为O(nlogn)O(nlogn)O(nlogn).

分解：按左右进行/2/2/2的划分下去。

合并：由于底部的左右段已经有序，所以可以直接合并。

归返：就是递归到一个数时返回上一层，然后一层一层回去。

在这里，先贴一个板子：

void qsort(int a,int b)//左右
{	
    if(a==b)return;//递归到一个数返回
	int mid=(a+b)>>1;//划分中心
	int i=a,j=mid+1,k=a;
    qsort(a,mid),qsort(mid+1,b);左右递归
    while(i<=mid&&j<=b)//有序合并
	if(c[i]<=c[j])
	  {
	  	d[k++]=c[i++];
	  }
	else 
	  {
	  	d[k++]=c[j++];
	  	ans+=mid-i+1;//求逆序对
	  }
	while(i<=mid)//处理没填完的
	  	d[k++]=c[i++];
	while(j<=b)
	    d[k++]=c[j++];
	for(int l=a;l<=b;l++)//操作完赋值
	    c[l]=d[l];
}

void insert(string s,int x)
{
    int len=s.size()-1;
    int las=0,w;
    for(rr int i=0;i<=len;i++)
    {
        w=idx(s[i]);
        if(ch[las][w]==0) 
         ch[las][w]=++tot;
        las=ch[las][w];

    val[las]=x;

int find(string s)

        las=ch[las][w];

}
